Here's a nifty extension method which will let you get the current index when iterating over a collection.

///

/// Provides a current index within a foreach.
///

/// The type of object in the collection
/// The collection to iterate over
/// Delegate to perform on each item
public static void Each(this IEnumerable collection, Action action)
{
var i = 0;
foreach (var item in collection) action(item, i++);
}
