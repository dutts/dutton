http://randypatterson.com/2007/09/how-to-design-a-fluent-interface/

 

public class Thing
{
private readonly ThingFluentInterface _set;
public ThingFluentInterface Set
{
get
{
return _set;
}
}

public string Name { get; set; }
public int Age { get; set; }

public Thing()
{
Name = "Default Name";
Age = 99;
_set = new ThingFluentInterface(this);
}

public class ThingFluentInterface
{
private readonly Thing _thing;

public ThingFluentInterface(Thing thing)
{
_thing = thing;
}

public ThingFluentInterface Name(string name)
{
_thing.Name = name;
return this;
}

public ThingFluentInterface Age(int age)
{
_thing.Age = age;
return this;
}
}
}
